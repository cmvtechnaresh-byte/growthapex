// HIGH-COMPATIBILITY GOOGLE APPS SCRIPT CODE

function doPost(e) {
  var lock = LockService.getScriptLock();
  lock.tryLock(10000); // 10s wait for previous processes
  
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var data = {};
    
    // Attempt to parse data from all possible sources (form or json)
    if (e.postData && e.postData.contents) {
      try {
        data = JSON.parse(e.postData.contents);
      } catch (err) {
        // Not JSON, use parameters instead
        data = e.parameter;
      }
    } else {
      data = e.parameter;
    }
    
    // Map data fields
    var type = data.type || 'lead';
    var email = data.email || 'N/A';
    var phone = data.phone || 'N/A';
    var niche = data.niche || 'N/A';
    var adSpend = data.adSpend || 'N/A';
    var socials = data.socials || 'N/A';
    var revenueGoal = data.revenueGoal || 'N/A';
    var timestamp = new Date();

    var sheetName = type === 'newsletter' ? 'Newsletter' : 'Leads';
    var targetSheet = ss.getSheetByName(sheetName);
    
    if (!targetSheet) {
      targetSheet = ss.insertSheet(sheetName);
      if (type === 'newsletter') {
        targetSheet.appendRow(['Timestamp', 'Email']);
      } else {
        targetSheet.appendRow(['Timestamp', 'Email', 'Phone', 'Niche', 'Ad Spend', 'Socials', 'Revenue Goal']);
      }
      targetSheet.getRange(1, 1, 1, targetSheet.getLastColumn()).setFontWeight("bold").setBackground("#e0e0e0");
    }
    
    if (type === 'newsletter') {
      targetSheet.appendRow([timestamp, email]);
    } else {
      targetSheet.appendRow([timestamp, email, phone, niche, adSpend, socials, revenueGoal]);
    }
    
    return ContentService.createTextOutput(JSON.stringify({"result": "success"}))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({"result": "error", "message": error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  } finally {
    lock.releaseLock();
  }
}

function doGet(e) {
  return ContentService.createTextOutput("GrowthApex Connect is LIVE. Use POST method to send leads.");
}
